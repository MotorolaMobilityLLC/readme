PPPS29.55-25-2 For Moto G6 Play

Kernel Modules:
---------------
Download prebuilts folder from Android distribution. Move it to $my_top_dir

For example:

git clone https://android.googlesource.com/platform/prebuilts/gcc/linux-x86/arm/arm-eabi-4.8

mkdir -p  $my_top_dir/prebuilts/gcc/linux-x86/arm

mv arm-eabi-4.8 $my_top_dir/prebuilts/gcc/linux-x86/arm/.

git clone https://android.googlesource.com/platform/prebuilts/gcc/linux-x86/arm/arm-linux-androideabi-4.9

mkdir -p  $my_top_dir/prebuilts/gcc/linux-x86/arm

mv arm-linux-androideabi-4.9 $my_top_dir/prebuilts/gcc/linux-x86/arm/.

mkdir -p $my_top_dir/out/target/product/generic/obj/kernel/msm-3.18

kernel_out_dir=$my_top_dir/out/target/product/generic/obj/kernel/msm-3.18

bcross=$my_top_dir/prebuilts/gcc/linux-x86/arm/arm-eabi-4.8/bin/arm-eabi-
ccross=$my_top_dir/prebuilts/gcc/linux-x86/arm/arm-linux-androideabi-4.9/bin/arm-linux-androidkernel- 

( perl -le 'print \"# This file was automatically generated from:\n#\t\" . join(\"\n#\t\", @ARGV) . \"\n\"' kernel/msm-3.18/arch/arm/configs/msmcortex-perf_defconfig kernel/msm-3.18/arch/arm/configs/ext_config/moto-msmcortex.config kernel/msm-3.18/arch/arm/configs/ext_config/mot8937.config kernel/msm-3.18/arch/arm/configs/ext_config/mot8937-jeter.config kernel/msm-3.18/arch/arm/configs/ext_config/mot8937-aljeter.config && cat kernel/msm-3.18/arch/arm/configs/msmcortex-perf_defconfig kernel/msm-3.18/arch/arm/configs/ext_config/moto-msmcortex.config kernel/msm-3.18/arch/arm/configs/ext_config/mot8937.config kernel/msm-3.18/arch/arm/configs/ext_config/mot8937-jeter.config kernel/msm-3.18/arch/arm/configs/ext_config/mot8937-aljeter.config ) > $kernel_out_dir/mapphone_defconfig || ( rm -f $kernel_out_dir/mapphone_defconfig && false )

cp $kernel_out_dir/mapphone_defconfig $kernel_out_dir/.config 

make -j16 -C kernel/msm-3.18 O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$ccross KBUILD_BUILD_USER= KBUILD_BUILD_HOST= defoldconfig

make -j16 -C kernel/msm-3.18 O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$ccross KBUILD_BUILD_USER= KBUILD_BUILD_HOST= headers_install

make -j16 -C kernel/msm-3.18 O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$ccross KBUILD_BUILD_USER= KBUILD_BUILD_HOST= defoldconfig

make -j16 -C kernel/msm-3.18 O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$ccross oldconfig

make -j16 -C kernel/msm-3.18 KBUILD_RELSRC=$my_top_dir/kernel/msm-3.18 O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$ccross KBUILD_BUILD_USER= KBUILD_BUILD_HOST= KCFLAGS=-mno-android

make -j16 -C kernel/msm-3.18 KBUILD_RELSRC=$my_top_dir/kernel/msm-3.18 O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$ccross KBUILD_BUILD_USER= KBUILD_BUILD_HOST= KCFLAGS=-mno-android dtbs

make -j16 -C kernel/msm-3.18 KBUILD_RELSRC=$my_top_dir/kernel/msm-3.18 O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$ccross KBUILD_BUILD_USER= KBUILD_BUILD_HOST= KCFLAGS=-mno-android modules

make -j16 -C kernel/msm-3.18 KBUILD_RELSRC=$my_top_dir/kernel/msm-3.18 O=$kernel_out_dir INSTALL_MOD_PATH=$my_top_dir/out/target/product/generic INSTALL_MOD_STRIP="--strip-debug --remove-section=.note.gnu.build-id" ARCH=arm CROSS_COMPILE=$ccross KBUILD_BUILD_USER= KBUILD_BUILD_HOST= modules_install

make -C kernel/msm-3.18 O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$bcross clean

make -C kernel/msm-3.18 O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$bcross mrproper

make -C kernel/msm-3.18 O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$bcross distclean

make -j16 -C kernel/msm-3.18 O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$ccross KBUILD_BUILD_USER= KBUILD_BUILD_HOST= tags

env KCONFIG_NOTIMESTAMP=true make -j16 -C kernel/msm-3.18 O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$ccross KBUILD_BUILD_USER= KBUILD_BUILD_HOST= menuconfig

env KCONFIG_NOTIMESTAMP=true make -j16 -C kernel/msm-3.18 O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$ccross KBUILD_BUILD_USER= KBUILD_BUILD_HOST= savedefconfig

WLAN Driver:
------------

cd $PWD/out/target/product/generic/obj

make -j16 -C kernel/msm-3.18 M=$my_top_dir/vendor/qcom/opensource/wlan/prima O=$kernel_out_dir ARCH=arm CROSS_COMPILE=$ccross KCFLAGS=-mno-android modules WLAN_ROOT=$my_top_dir/vendor/qcom/opensource/wlan/prima MODNAME=wlan BOARD_PLATFORM=msm8937 CONFIG_PRONTO_WLAN=m
